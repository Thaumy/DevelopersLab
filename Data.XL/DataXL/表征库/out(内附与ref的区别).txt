using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApplication4
{
    class Program
    {
        static void Main(string[] args)
        {
            int value;
            String str1, str2;
            Console.WriteLine("显示输出");
            Method(out value, out str1, out str2);
            Console.WriteLine(">{0}-{1}-{2}<", value,str1,str2);
            Console.ReadKey();

        }

        static void Method(out int ex0, out string ex1, out string ex2)
        {

            ex0 = 44;
            ex1 = "SS1";
            ex2 = null;

        }
        
        /*
out关键字与ref关键字相似，

        也是声明引用传递参数的，

        但ref要求变量必须在传递之前对实在参数初始化。

        尽管作为out参数传递的变量不需要在传递之前初始化，

        但需要调用方法在方法返回之前对out传递的参数赋值
*/

    }
}
