using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace StringBuilder
{
    class Program
    {
        static void Main(string[] args)
        {
            System.Text.StringBuilder str1 = new System.Text.StringBuilder();

            str1.Append("TEXT1");
            Console.Write(str1);
            Console.WriteLine();
            Console.ReadLine();

            str1.AppendLine("TEXT2");
            Console.Write(str1);
            Console.ReadLine();

            str1.Append("TEXT3");
            Console.Write(str1.ToString());
            Console.WriteLine();
            Console.ReadLine();

            str1.AppendLine("TEXT4");
            Console.Write(str1.ToString());
            Console.ReadLine();


        }
    }
}
