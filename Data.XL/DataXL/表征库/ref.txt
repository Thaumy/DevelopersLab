using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApplication3
{
    class BUG
    {
        static void Main(string[] args)
        {

            int num1 = 3;
            int num2 = 8;
            Console.WriteLine("SS");
            Console.WriteLine("交换之前的顺序是{0}和{1}", num1, num2);
            BUG BUG1 = new BUG();
            BUG1.Swap(ref num1,ref num2);
            Console.WriteLine("交换之后的顺序是{0}和{1}", num1, num2);
            Console.ReadKey();


        }

        private void Swap(ref int num1,ref int num2)
        {
            int temp;
            temp = num1;
            num1 = num2;
            num2 = temp;

        }
    }
}
