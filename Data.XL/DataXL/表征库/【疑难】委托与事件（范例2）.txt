using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApplication1
{
    class Program
    {
        static void Main(string[] args)
        {
            Main0 NewClass = new Main0();
            eventCauser EventClass = new eventCauser();//带有事件的类实例
            //↓客户对事件进行订阅
            EventClass.Event1 += NewClass.main;

            EventClass.input();

        }

        public class eventCauser
        {
            string par;

            public delegate void myEventHandler(int param);
            public event myEventHandler Event1;

            public void input()
            {
                
                if ((this.par = Console.ReadLine())=="激发事件") { Event1(1); }
            }
        }

        public class Main0
        {
            public void main(int par2)
            {
                Console.WriteLine("客户1.已被激发，最后一次的 Event 参数为 {0}",par2);
            }
        }


    }
}
