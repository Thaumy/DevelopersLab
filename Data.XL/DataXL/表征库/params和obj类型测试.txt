using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace params和OBJ测试
{
    class Program
    {
        static void Main(string[] args)
        {
            p demo = new p();


            int[] temp = new int[10];
            temp[1] = 20;

            string d = "HI";
            object p = d;//object类型是所有类型的基类
            demo.a(temp);
            demo.c(p);

        }
        public class p
        {
            public void a(params int[] b)//合法的声明，params前缀可以不用传值
            {
                Console.WriteLine(b[1]);
            }
            public void c(object d)
            {
                Console.WriteLine(d);
            }
        }

    }
}
