using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApplication6
{
    class Program
    {
        static void Main(string[] args)
        {

            //实例化类Ep0
            EP0 Exameple0 = new EP0(78, 69);

            Console.WriteLine("x = {0} , y = {1}", Exameple0.x, Exameple0.y);

            Exameple0.doWork();
            Exameple0.eatFood();

            Exameple0.x = 1;
            Exameple0.y = 2;

            Console.WriteLine("x = {0} , y = {1}", Exameple0.x, Exameple0.y);
        }


        public interface IEP
        {
            int x//声明操作数x
            {
                get;
                set;
            }
            int y//声明操作数y
            {
                get;
                set;
            }

            void doWork();

            void eatFood();

            //接口内部只能对接口体声明，不能进行实例化


        }

        public class EP0 : IEP
        {
            public EP0(int x, int y)
            {
                opr_x = x;
                opr_y = y;
            }
            int opr_x;
            int opr_y;

            public int x
            {
                get { return opr_x; }
                set { opr_x = value; }
            }
            public int y
            {
                get { return opr_y; }
                set { opr_y = value; }
            }

            public void doWork()
            {
                Console.WriteLine("这里调用doWork()方法");
            }

            public void eatFood()
            {
                Console.WriteLine("这里调用eatFood()方法");
            }
        }
    }
}
