using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApplication
{
    class Program
    {
        static void Main(string[] args)
        {

            ClassE nC = new ClassE();
            nC.firstE += member1.outPut;

            nC.MainE(1);

        }
    }

    //用于触发事件的类
    public class ClassE
    {

        public delegate void firstEventHandler();
        public event firstEventHandler firstE;

        public void MainE(int opr)
        {
            if (opr==1) { firstE(); }
        }
    }

    public class member1
    {
        public static void outPut()
        {
            Console.WriteLine("客户1.已经被事件激活");
        }
    }
}
